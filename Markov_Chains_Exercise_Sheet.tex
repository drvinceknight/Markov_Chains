\begin{itemize}
	\item If a student is Rich, in the next time step the student will be:
	\begin{itemize}
		\item Average: .75
		\item Poor: .2
		\item In Debt: .05
	\end{itemize}
	\item If a student is Average, in the next time step the student will be:
	\begin{itemize}
		\item Rich: .05
		\item Average: .2
		\item In Debt: .45
	\end{itemize}
	\item If a student is Poor, in the next time step the student will be:
	\begin{itemize}
		\item Average: .4
		\item Poor: .3
		\item In Debt: .2
	\end{itemize}
	\item If a student is In Debt, in the next time step the student will be:
	\begin{itemize}
		\item Average: .15
		\item Poor: .3
		\item In Debt: .55
	\end{itemize}
\end{itemize}
